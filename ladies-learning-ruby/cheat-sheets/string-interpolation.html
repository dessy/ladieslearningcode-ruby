String Interpolation
> "The #{"cat"} in the hat." > animal = "cat"
> "The #{animal} in the hat."
> "#{99} bottles of beer on the wall!" > amount = 99
> "#{amount} bottles of beer on the wall!"
> amount = 2
> 'Using #{amount} with single quotes.' 		> amount = 2
> 'Adding ' + "#{amount} strings together"
> one = 1
> "#{one} plus #{one} is equal to #{one + one}" 		> one = 1
> two = 2
> "#{one} times #{two} is equal to #{one * two}"
